function extractClothesIdHint(hit = {}) {
  const fileName = normalizeText(hit.fileName)
  const snippet = normalizeText(hit.snippet)

  const byFileName = fileName.replace(/\.[^.]+$/, '')
  if (/^[a-f0-9]{24,32}$/i.test(byFileName)) {
    return byFileName
  }

  const match = snippet.match(/clothes_id\s*:\s*([a-zA-Z0-9_-]+)/i)
  if (match && match[1]) {
    return match[1].trim()
  }

  return ''
}

function safeParseRecommendationText(rawText = '') {
  const text = normalizeText(rawText)
  if (!text) {
    return {
      summary: '',
      replyText: '',
      outfitLines: [],
      tips: [],
      selectedClothesIds: []
    }
  }

  try {
    const parsed = JSON.parse(extractJsonObjectString(text))
    return {
      summary: normalizeText(parsed.summary),
      replyText: normalizeText(parsed.replyText),
      outfitLines: normalizeStringList(parsed.outfitLines),
      tips: normalizeStringList(parsed.tips),
      selectedClothesIds: normalizeStringList(parsed.selectedClothesIds)
    }
  } catch (error) {
    logWarning('knowledgeRetrievalParser.safeParseRecommendationText', 'json parse failed', {
      message: error.message
    })
    return {
      summary: '',
      replyText: text,
      outfitLines: [],
      tips: [],
      selectedClothesIds: []
    }
  }
}

function resolveMatchedClothesFromRetrieval({ clothesList = [], aiPayload = {} } = {}) {
  const byId = new Map()
  const byDocId = new Map()
  const byFileId = new Map()

  clothesList.forEach(item => {
    const id = normalizeText(item && item._id)
    const docId = normalizeText(item && (item.bailian_doc_id || item.knowledge_doc_id))
    const fileId = normalizeText(item && item.bailian_file_id)

    if (id) byId.set(id, item)
    if (docId) byDocId.set(docId, item)
    if (fileId) byFileId.set(fileId, item)
  })

  const result = []
  const pushUnique = item => {
    if (item && !result.some(current => current._id === item._id)) {
      result.push(item)
    }
  }

  ;(aiPayload.selectedClothesIds || []).forEach(id => pushUnique(byId.get(id)))

  ;(aiPayload.retrievalItems || []).forEach(hit => {
    pushUnique(byDocId.get(normalizeText(hit.docId)))
    pushUnique(byFileId.get(normalizeText(hit.fileId)))
    pushUnique(byId.get(normalizeText(hit.clothesIdHint)))
  })

  return result
}
